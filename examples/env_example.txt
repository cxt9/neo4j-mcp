# Neo4j MCP Configuration Example
# Copy this file to .env and adjust the values for your setup

# Basic Connection Settings (required)
NEO4J_HOST=localhost
NEO4J_PORT=7687
NEO4J_HTTP_PORT=7474

# Authentication (optional - leave empty for auth-disabled databases)
NEO4J_USERNAME=neo4j
NEO4J_PASSWORD=your-password-here

# Database Settings
NEO4J_DATABASE=neo4j
NEO4J_URI_SCHEME=bolt

# Security Settings
NEO4J_ENCRYPTED=false

# Connection Pool Settings (advanced)
# NEO4J_MAX_CONNECTION_LIFETIME=300
# NEO4J_MAX_CONNECTION_POOL_SIZE=100
# NEO4J_CONNECTION_TIMEOUT=30.0
